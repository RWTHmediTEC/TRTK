00448 TRTK::Coordinate<T> orthogonalMatrixToQuaternion(const Eigen::Matrix<T, 3, 3> & matrix)
00449 {
00450     // References:
00451     //
00452     // [1] http://www.euclideanspace.com/maths/geometry/rotations/conversions/matrixToQuaternion/index.htm
00453     // [2] http://en.wikipedia.org/wiki/Rotation_matrix
00454 
00455     // Note: Coordinate<T> is of the form (x, y, z, w) whereas quaternions are often notated (w, x, y, z)!
00456 
00457     using std::sqrt;
00458 
00459     TRTK::Coordinate<T> quaternion(0, 0, 0, 0);
00460 
00461     T trace = matrix.trace();
00462 
00463     if (trace > -1) // check because of the computation of the root
00464     {
00465         double s = 0.5 / sqrt(trace + 1);
00466         quaternion.x() = 0.25 / s;
00467         quaternion.y() = (matrix(2,1) - matrix(1,2)) * s;
00468         quaternion.z() = (matrix(0,2) - matrix(2,0)) * s;
00469         quaternion.w() = (matrix(1,0) - matrix(0,1)) * s;
00470     }
00471     else if (matrix(0,0) > matrix(1,1) && matrix(0,0) > matrix(2,2))
00472     {
00473         double s = 2.0 * sqrt(1 + matrix(0,0) - matrix(1,1) - matrix(2,2));
00474         quaternion.x() = (matrix(2,1) - matrix(1,2)) / s;
00475         quaternion.y() = 0.25 * s;
00476         quaternion.z() = (matrix(0,1) + matrix(1,0)) / s;
00477         quaternion.w() = (matrix(0,2) + matrix(2,0)) / s;
00478     }
00479     else if (matrix(1,1) > matrix(2,2))
00480     {
00481         double s = 2 * sqrt(1 + matrix(1,1) - matrix(0,0) - matrix(2,2));
00482         quaternion.x() = (matrix(0,2) - matrix(2,0)) / s;
00483         quaternion.y() = (matrix(0,1) + matrix(1,0)) / s;
00484         quaternion.z() = 0.25 * s;
00485         quaternion.w() = (matrix(1,2) + matrix(2,1)) / s;
00486     }
00487     else
00488     {
00489         double s = 2 * sqrt(1 + matrix(2,2) - matrix(0,0) - matrix(1,1));
00490         quaternion.x() = (matrix(1,0) - matrix(0,1)) / s;
00491         quaternion.y() = (matrix(0,2) + matrix(2,0)) / s;
00492         quaternion.z() = (matrix(1,2) + matrix(2,1)) / s;
00493         quaternion.w() = 0.25 * s;

00521 template <class T>
00522 TRTK::Coordinate<T> orthogonalMatrixToQuaternion2(const Eigen::Matrix<T, 3, 3> & matrix)
00523 {
00524     // See [1] for more details about the implementation.
00525 
00526     // The matrix K3 is selfadjoint because it is real and symmetric. Thus we can use an
00527     // appropriate solver. The solver employed below only uses the lower triangular part
00528     // of the matrix hence we can leave the upper part uninitialized.
00529 
00530     typedef Eigen::Matrix<T, 4, 4> Matrix4T;
00531     typedef Eigen::Matrix<T, 4, 1> Vector4T;
00532 
00533     Matrix4T K3;
00534 
00535     K3(0,0) = matrix(0,0) - matrix(1,1) - matrix(2,2);
00536     K3(1,0) = matrix(1,0) + matrix(0,1);
00537     K3(2,0) = matrix(2,0) + matrix(0,2);
00538     K3(3,0) = matrix(1,2) - matrix(2,1);
00539     K3(1,1) = matrix(1,1) - matrix(0,0) - matrix(2,2);
00540     K3(2,1) = matrix(2,1) + matrix(1,2);
00541     K3(3,1) = matrix(2,0) - matrix(0,2);
00542     K3(2,2) = matrix(2,2) - matrix(0,0) - matrix(1,1);
00543     K3(3,2) = matrix(0,1) - matrix(1,0);
00544     K3(3,3) = matrix(0,0) + matrix(1,1) + matrix(2,2);
00545 
00546     K3 /= 3;
00547 
00548     Eigen::SelfAdjointEigenSolver<Matrix4T> solver(K3);
00549 
00550     Vector4T quaternion = solver.eigenvectors().col(3).normalized();
00551 
00552     return Coordinate<T>(quaternion);
00553 }

00566 template <class T>
00567 Eigen::Matrix<T, 3, 3> quaternionToOrthogonalMatrix(T q0, T q1, T q2, T q3)
00568 {
00569     assert(abs(q0 * q0 + q1 * q1 + q2 * q2 + q3 * q3 - 1) < 1e-5); // check for unit quaternion
00570 
00571     Eigen::Matrix<T, 3, 3> R;
00572 
00573     R(0, 0) = 1 - 2 * (q2 * q2 + q3 * q3);
00574     R(0, 1) = 2 * q1 * q2 - 2 * q0 * q3;
00575     R(0, 2) = 2 * q1 * q3 + 2 * q0 * q2;
00576     R(1, 0) = 2 * q1 * q2 + 2 * q0 * q3;
00577     R(1, 1) = 1 - 2 * (q1 * q1  + q3 * q3);
00578     R(1, 2) = 2 * q2 * q3 - 2 * q0 * q1;
00579     R(2, 0) = 2 * q1 * q3 - 2 * q0 * q2;
00580     R(2, 1) = 2 * q2 * q3 + 2 * q0 * q1;
00581     R(2, 2) = 1 - 2 * (q1 * q1 + q2 * q2);
00582 
00583     return R;

00691 template <class T>
00692 T randn()
00693 {
00694     /*
00695     This algorithm uses the Marsaglia polar method. For more details, please
00696     have a look at "A convenient method for generating normal variables" by
00697     G. Marsaglia and T. A. Bray (SIAM Rev. 6, 260-264, 1964).
00698     */
00699 
00700     // The polar method yields two random variables per computation. So one
00701     // random variable is cached each time and returned on the next call.
00702 
00703     using std::log;
00704     using std::sqrt;
00705 
00706     static bool cache_is_valid = false;
00707     static T cached_random_variable = 0;
00708 
00709     if (!cache_is_valid)
00710     {
00711         T v1, v2, s;
00712 
00713         do
00714         {
00715             v1 = rand<T>(-1, 1);
00716             v2 = rand<T>(-1, 1);
00717 
00718             s = v1 * v1 + v2 * v2;
00719         }
00720         while (s >= 1 || s == 0);
00721 
00722         const T t = sqrt(-2 * log(s) / s);
00723 
00724         T random_variable = v1 * t;
00725 
00726         cached_random_variable = v2 * t;
00727         cache_is_valid = true;
00728 
00729         return random_variable;
00730     }
00731     else
00732     {
00733         cache_is_valid = false;
00734 
00735         return cached_random_variable;
00736     }
00737 }

00878 inline ValueType standardDeviation(const Container & container, ValueType null_value = ValueType())
00879 {
00880     size_t n = container.size();
00881     assert (n >= 0);
00882 
00883     if (n == 0) return null_value;
00884 
00885     ValueType mean = Tools::mean(container, null_value);
00886     ValueType variance = null_value;
00887 
00888     for (typename Container::const_iterator it = container.begin(); it != container.end(); ++it)
00889     {
00890         variance = variance + (*it - mean) * (*it - mean);
00891     }
