00450 TRTK::Coordinate<T> orthogonalMatrixToQuaternion(const Eigen::Matrix<T, 3, 3> & matrix)
00451 {
00452     // References:
00453     //
00454     // [1] http://www.euclideanspace.com/maths/geometry/rotations/conversions/matrixToQuaternion/index.htm
00455     // [2] http://en.wikipedia.org/wiki/Rotation_matrix
00456 
00457     // Note: Coordinate<T> is of the form (x, y, z, w) whereas quaternions are often notated (w, x, y, z)!
00458 
00459     using std::sqrt;
00460 
00461     TRTK::Coordinate<T> quaternion(0, 0, 0, 0);
00462 
00463     T trace = matrix.trace();
00464 
00465     if (trace > -1) // check because of the computation of the root
00466     {
00467         double s = 0.5 / sqrt(trace + 1);
00468         quaternion.x() = 0.25 / s;
00469         quaternion.y() = (matrix(2,1) - matrix(1,2)) * s;
00470         quaternion.z() = (matrix(0,2) - matrix(2,0)) * s;
00471         quaternion.w() = (matrix(1,0) - matrix(0,1)) * s;
00472     }
00473     else if (matrix(0,0) > matrix(1,1) && matrix(0,0) > matrix(2,2))
00474     {
00475         double s = 2.0 * sqrt(1 + matrix(0,0) - matrix(1,1) - matrix(2,2));
00476         quaternion.x() = (matrix(2,1) - matrix(1,2)) / s;
00477         quaternion.y() = 0.25 * s;
00478         quaternion.z() = (matrix(0,1) + matrix(1,0)) / s;
00479         quaternion.w() = (matrix(0,2) + matrix(2,0)) / s;
00480     }
00481     else if (matrix(1,1) > matrix(2,2))
00482     {
00483         double s = 2 * sqrt(1 + matrix(1,1) - matrix(0,0) - matrix(2,2));
00484         quaternion.x() = (matrix(0,2) - matrix(2,0)) / s;
00485         quaternion.y() = (matrix(0,1) + matrix(1,0)) / s;
00486         quaternion.z() = 0.25 * s;
00487         quaternion.w() = (matrix(1,2) + matrix(2,1)) / s;
00488     }
00489     else
00490     {
00491         double s = 2 * sqrt(1 + matrix(2,2) - matrix(0,0) - matrix(1,1));
00492         quaternion.x() = (matrix(1,0) - matrix(0,1)) / s;
00493         quaternion.y() = (matrix(0,2) + matrix(2,0)) / s;
00494         quaternion.z() = (matrix(1,2) + matrix(2,1)) / s;
00495         quaternion.w() = 0.25 * s;
00496     }
00497

00523 template <class T>
00524 TRTK::Coordinate<T> orthogonalMatrixToQuaternion2(const Eigen::Matrix<T, 3, 3> & matrix)
00525 {
00526     // See [1] for more details about the implementation.
00527 
00528     // The matrix K3 is selfadjoint because it is real and symmetric. Thus we can use an
00529     // appropriate solver. The solver employed below only uses the lower triangular part
00530     // of the matrix hence we can leave the upper part uninitialized.
00531 
00532     typedef Eigen::Matrix<T, 4, 4> Matrix4T;
00533     typedef Eigen::Matrix<T, 4, 1> Vector4T;
00534 
00535     Matrix4T K3;
00536 
00537     K3(0,0) = matrix(0,0) - matrix(1,1) - matrix(2,2);
00538     K3(1,0) = matrix(1,0) + matrix(0,1);
00539     K3(2,0) = matrix(2,0) + matrix(0,2);
00540     K3(3,0) = matrix(1,2) - matrix(2,1);
00541     K3(1,1) = matrix(1,1) - matrix(0,0) - matrix(2,2);
00542     K3(2,1) = matrix(2,1) + matrix(1,2);
00543     K3(3,1) = matrix(2,0) - matrix(0,2);
00544     K3(2,2) = matrix(2,2) - matrix(0,0) - matrix(1,1);
00545     K3(3,2) = matrix(0,1) - matrix(1,0);
00546     K3(3,3) = matrix(0,0) + matrix(1,1) + matrix(2,2);
00547 
00548     K3 /= 3;
00549 
00550     Eigen::SelfAdjointEigenSolver<Matrix4T> solver(K3);
00551 
00552     Vector4T quaternion = solver.eigenvectors().col(3).normalized();
00553 
00554     return Coordinate<T>(quaternion);
00555 }

00568 template <class T>
00569 Eigen::Matrix<T, 3, 3> quaternionToOrthogonalMatrix(T q0, T q1, T q2, T q3)
00570 {
00571     // Revised code.
00572 
00573     // Identical to what you can find in the Wikipedia article
00574     // https://en.wikipedia.org/wiki/Quaternions_and_spatial_rotation
00575     // This matrix is a left-handed (post-multiplied) rotation matrix.
00576 
00577     assert(abs(q0 * q0 + q1 * q1 + q2 * q2 + q3 * q3 - 1) < 1e-5); // check for unit quaternion
00578 
00579     Eigen::Matrix<T, 3, 3> R;
00580 
00581     R(0, 0) = 1 - 2 * (q2 * q2 + q3 * q3);
00582     R(0, 1) = 2 * q1 * q2 - 2 * q0 * q3;
00583     R(0, 2) = 2 * q1 * q3 + 2 * q0 * q2;
00584 
00585     R(1, 0) = 2 * q1 * q2 + 2 * q0 * q3;
00586     R(1, 1) = 1 - 2 * (q1 * q1  + q3 * q3);
00587     R(1, 2) = 2 * q2 * q3 - 2 * q0 * q1;
00588 
00589     R(2, 0) = 2 * q1 * q3 - 2 * q0 * q2;
00590     R(2, 1) = 2 * q2 * q3 + 2 * q0 * q1;
00591     R(2, 2) = 1 - 2 * (q1 * q1 + q2 * q2);
00592 
00593     return R;
00594 }

00708 template <class T>
00709 T randn()
00710 {
00711     /*
00712     This algorithm uses the Marsaglia polar method. For more details, please
00713     have a look at "A convenient method for generating normal variables" by
00714     G. Marsaglia and T. A. Bray (SIAM Rev. 6, 260-264, 1964).
00715     */
00716 
00717     // The polar method yields two random variables per computation. So one
00718     // random variable is cached each time and returned on the next call.
00719 
00720     using std::log;
00721     using std::sqrt;
00722 
00723     static bool cache_is_valid = false;
00724     static T cached_random_variable = 0;
00725 
00726     if (!cache_is_valid)
00727     {
00728         T v1, v2, s;
00729 
00730         do
00731         {
00732             v1 = rand<T>(-1, 1);
00733             v2 = rand<T>(-1, 1);
00734 
00735             s = v1 * v1 + v2 * v2;
00736         }
00737         while (s >= 1 || s == 0);
00738 
00739         const T t = sqrt(-2 * log(s) / s);
00740 
00741         T random_variable = v1 * t;
00742 
00743         cached_random_variable = v2 * t;
00744         cache_is_valid = true;
00745 
00746         return random_variable;
00747     }
00748     else
00749     {
00750         cache_is_valid = false;
00751 
00752         return cached_random_variable;
00753     }
00754 }

00895 inline ValueType standardDeviation(const Container & container, ValueType null_value = ValueType())
00896 {
00897     size_t n = container.size();
00898     assert (n >= 0);
00899 
00900     if (n == 0) return null_value;
00901 
00902     ValueType mean = Tools::mean(container, null_value);
00903     ValueType variance = null_value;
00904 
00905     for (typename Container::const_iterator it = container.begin(); it != container.end(); ++it)
00906     {
00907         variance = variance + (*it - mean) * (*it - mean);
00908     }
00909 
00910     double c4 = 1.0 - 1.0/(4.0*n) - 7.0/(32.0*n*n) - 19.0/(128.0*n*n*n);
00911 
00912     using std::sqrt;
00913 
00914     return sqrt(variance / (n - 1)) / c4;
