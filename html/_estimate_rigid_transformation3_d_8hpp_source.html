00392     /*
00393 
00394     Explanation of the algorithm (see also "Image Guided Therapy" lecture):
00395 
00396     We try to estimate the rotation around the centroids of both data sets. Thus we first
00397     subtract the mean of both point sets such that the sets are centered around the point
00398     of origin. Then we compute the correlation matrix between both point sets:
00399 
00400     R_XY = (X - mean(X)) * (Y - mean(Y))^T.
00401 
00402     By the way, R_XY is nothing else but the cross-covariance matrix of X and Y, cov(X,Y).
00403     Now, R_XY can be (equivalently) decomposed in two ways, first by means of a singular
00404     value decomposition (SVD)
00405 
00406     R_XY = U * Sigma * V^T,
00407 
00408     where U and V are orthogonal matrices and Sigma is a diagonal matrix, and second
00409     by using an Eigen decomposition
00410 
00411     R_XY = (X - mean(X)) * (Y - mean(Y))^T
00412          = x * y^T
00413          = x * [R * x]^T                (use y = R * x)
00414          = x * x^T * R^T
00415          = S * D * S^T * R^T            (Eigen decomposition of x * x^T)
00416 
00417     where x and y are the zero mean vectors of X and Y, respectively, D is a diagonal
00418     matrix, S is an orthogonal matrix and R is the sought rotation matrix. Comparison of
00419 
00420     R_XY = U * Sigma * V^T,
00421          = S * D     * S^T * R^T
00422 
00423     yields
00424 
00425     V^T = S^T * R^T = U^T * R^T.
00426 
00427     It follows that
00428 
00429     R = V * U^T.
00430 
00431     Now we can construct the homogeneous transformation matrix as follows:
00432 
00433     | R | T |     | I | mean(Y) |     | R | 0 |     | I | -mean(X) |
00434     | --+-- |  =  | --+-------- |  *  | --+-- |  *  | --+--------- |
00435     | 0 | 1 |     | 0 |   1     |     | 0 | 1 |     | 0 |    1     |
00436 
00437                   | R | mean(Y) - R * mean(X) |
00438                =  | --+---------------------- |
00439                   | 0 |          1            |
00440 
00441     Note: The above equation basically says: move the centroid of the source data to
00442           the point of origin, rotate the data and move it to the position of the
00443           centroid of the target data.
00444 
00445     Note: I is the identity matrix and T is an arbitrary translation vector.
00446 
00447     Note: Actually the rotation matrix on the left hand side should have had a name
00448           different from R, but since the entries are the same, we kept the name.
00449     */
00450 
00451     MatrixXT X = super::m_source_points;
00452     MatrixXT Y = super::m_target_points;
00453 
00454     // subtract mean
00455 
00456     Vector3T mean_X = X.rowwise().mean();
00457     Vector3T mean_Y = Y.rowwise().mean();
00458 
00459     X.colwise() -= mean_X;
00460     Y.colwise() -= mean_Y;
00461 
00462     // compute SVD of cross-covariance matrix
00463 
00464     MatrixXT R_XY = X * Y.adjoint();
00465 
00466     Eigen::JacobiSVD<MatrixXT> svd(R_XY, Eigen::ComputeThinU | Eigen::ComputeThinV);
00467 
00468     // compute estimate of the rotation matrix
00469 
00470     Matrix3T R = svd.matrixV() * svd.matrixU().adjoint();
00471 
00472     // assure a right-handed coordinate system
00473 
00474     if (R.determinant() < 0)
00475     {
00476         R = svd.matrixV() * Vector3T(1, 1, -1).asDiagonal() * svd.matrixU().transpose();
00477     }
00478 
00479     // construct homogeneous transformation matrix
00480 
00481     super::m_transformation_matrix.block(0, 0, 3, 3) = R;
00482     super::m_transformation_matrix.block(0, 3, 3, 1) = mean_Y - R * mean_X;
00483     super::m_transformation_matrix.block(3, 0, 1, 3) = RowVector3T::Zero();
00484     // super::m_transformation_matrix(3, 3) = 1.0; // already one from initialisation
00485 }
