00309 {
00310     IplImage * output = cvCreateImage(cvGetSize(input), IPL_DEPTH_32F, 1);
00311 
00312     // Is input an RGB color image?
00313

00315     {
00316         // Convert image into a gray scale image...
00317 
00318         IplImage * grayScaleImage = cvCreateImage(cvGetSize(input), IPL_DEPTH_8U, 1);
00319         cvCvtColor(input, grayScaleImage, CV_RGB2GRAY);
00320 
00321         // ... and convert its pixels into doubles.

00323         cvConvertScale(grayScaleImage, output, 1.0/255);
00324 
00325         // Release resources.

00329     else if (input->nChannels == 1 && (input->depth == IPL_DEPTH_8U || input->depth == IPL_DEPTH_8S))
00330     {
00331         // Already grayscale. Thus, only convert its 8-bit pixels into doubles.

00333         cvConvertScale(input, output, 1.0/255);
00334     }

00336     {

00338         error.setErrorMessage("Can't handle image type.");
