00316 template <class T>
00317 T PivotCalibrationCombinatorialApproach<T>::compute()
00318 {
00319     /*
00320 
00321     Explanation of the algorithm
00322 
00323     If a tool touches the pivot point with its tip, the pivot point's location M
00324     in global coordinates is
00325 
00326     M = R_i * t + t_i
00327 
00328     where 'R_i' is the rotation and 't_i' the location of the tool sensor in the world or
00329     tracking coordinate system. 't' is the sought location of the tool tip in the sensor's
00330     local coordinate system. Note, that 't' remains the same for all measurements! Now,
00331     using two different measurements, the pivot point M can be eliminated
00332 
00333     R_i * t + t_i = R_j * t + t_j
00334 
00335     This can be rearranged to
00336 
00337     (R_i - R_j) * t = (t_j - t_i)
00338 
00339     Doing this for all combinations of all measurements yields a system of equations
00340     Ax = b (with x = t) which can be solved for the sought translation 't'.
00341 
00342     | R1 - R2   |         | t2   - t1 |
00343     | R1 - R3   |         | t3   - t1 |
00344     | R1 - R4   | * t  =  | t4   - t1 |      <==>    At = b
00345     |   ...     |         |     ...   |
00346     | Rn - Rn-1 |         | tn-1 - tn |
00347 
00348     t = [A^T * A]^(-1) * A^T * B        (Moore-Penrose pseudoinverse)
00349 
00350     Note: Not all combinations are taken into account. For instance, the case i == j
00351     is omitted.
00352 
00353     */
00354 
00355     const int n = rotations.size();
00356     const int max_number_of_combinations = n * (n - 1);
00357     int current_index = 0; // i-th entry (or "row") in A and b
00358 
00359     // Build up A and b and solve the system of equations
00360 
00361     MatrixXT A(3 * max_number_of_combinations, 3);
00362     MatrixXT b(3 * max_number_of_combinations, 1);
00363 
00364     for (int i = 0; i < n; ++i)
00365     {
00366         for (int j = 0; j < n; j++)
00367         {
00368             if (i == j || rotations[i].isApprox(rotations[j]))
00369             {
00370                 continue;
00371             }
00372 
00373             A.block(current_index * 3, 0, 3, 3) = rotations[i] - rotations[j];
00374             b.block(current_index * 3, 0, 3, 1) = locations[j] - locations[i];
00375 
00376             ++current_index;
00377         }
00378 
00379     }
00380 
00381     A.resize(current_index * 3, 3);
00382     b.resize(current_index * 3, 1);
00383 
00384     local_pivot_point = (A.transpose() * A).inverse() * (A.transpose() * b);
00385 
00386     // Compute an averaged global pivot point
00387 
00388     global_pivot_point = Vector3T(0, 0, 0);
00389 
00390     for (int i = 0; i < n; ++i)
00391     {
00392         global_pivot_point += rotations[i] * local_pivot_point + locations[i];
00393     }
00394 
00395     global_pivot_point /= n;
00396 
00397     // Compute the RMSE
00398 
00399     using std::sqrt;
00400     rmse = T(0);
00401 
00402     for (int i = 0; i < n; ++i)
00403     {
00404         Vector3T noisy_center_point = rotations[i] * local_pivot_point + locations[i];
00405         rmse += (noisy_center_point - global_pivot_point).squaredNorm();
00406     }
00407 
00408     rmse = sqrt(rmse / n);
00409 
00410     return rmse;
00411 }

00608 template <class T>
00609 T PivotCalibrationPATM<T>::compute()
00610 {
00611     /*
00612 
00613     Explanation of the algorithm
00614 
00615     If a tool touches the pivot point with its tip, the pivot point's location M
00616     in global coordinates is
00617 
00618     M = R_i * t + t_i
00619 
00620     where 'R_i' is the rotation and 't_i' the location of the tool sensor in the world or
00621     tracking coordinate system. 't' is the sought location of the tool tip in the sensor's
00622     local coordinate system. Note, that 't' remains the same for all measurements! Now,
00623     using two different measurements, the pivot point M can be eliminated
00624 
00625     R_i * t + t_i = R_j * t + t_j
00626 
00627     This can be rearranged to
00628 
00629     t = R_i^T * R_j * t + R_i^T * (t_j - t_i) =: phi(t)                               (*)
00630 
00631     This is actually the form of a fixed-point iteration, i.e. t_i+1 = phi(t_i). However,
00632     in this form phi is not a contraction mapping since the Lipschitz constant L is not
00633     strictly less than 1:
00634 
00635     || phi(x) - phi(y) || = || R_i^T * R_j * (x - y) || = L * || x - y||,   L = 1
00636 
00637     This can be changed by averaging at least two equations (*) with different
00638     measurements. Thus we construct phi(t) as follows
00639 
00640     phi(t) := R * t + p                                                              (**)
00641 
00642     where R := 1/n * sum_i R_i^T * R_(i+1),   p := 1/n * sum_i R_i^T * (t_(i+1) - t_i),
00643     and n is the number of measurements. Since the sum goes up to n, R_(n+1) and t_(n1+1)
00644     are set to R_1 and t_1, respectively.
00645 
00646     The equation (**) can also rewritten as
00647 
00648              | R  p |
00649     phi(t) = |      | * t = A * t
00650              | 0  1 |
00651 
00652     and thus t_i+1 = phi(t_i) = A * t_i = A^i * t0.
00653 
00654     */
00655 
00656     // Build up R and p
00657 
00658     Matrix3T R = Matrix3T::Zero();
00659     Vector3T p = Vector3T::Zero();
00660 
00661     const int n = rotations.size();
00662 
00663     for (int i = 0; i < n; ++i)
00664     {
00665         int k = i % n; // 0, 1, 2, ..., n-1, n
00666         int l = (i + 1) % n; // 1, 2, 3, ..., n, 0
00667 
00668         R += rotations[k].transpose() * rotations[l];
00669         p += rotations[k].transpose() * (locations[l] - locations[k]);
00670     }
00671 
00672     R /= n;
00673     p /= n;
00674 
00675     Matrix4T A = Matrix4T::Identity();
00676     A.block<3, 3>(0, 0) = R;
00677     A.block<3, 1>(0, 3) = p;
00678 
00679     // Carry out the fixed-point iterations
00680 
00681     Vector4T t = Vector4T(1, 1, 1, 1);
00682 
00683     if (number_of_iterations > 10)
00684     {
00685         // Compute the power of A
00686         Eigen::EigenSolver<MatrixXT> solver(A);
00687         Matrix4cT D = solver.eigenvalues().asDiagonal();
00688         Matrix4cT V = solver.eigenvectors();
00689         A = (V * D.array().pow(number_of_iterations).matrix() * V.inverse()).real();
00690         t = A * t;
00691     }
00692     else
00693     {
00694         for (unsigned i = 0; i < number_of_iterations; ++i)
00695         {
00696             t = A * t;
00697         }
00698     }
00699 
00700     local_pivot_point = t.head<3>();
00701 
00702     // Compute an averaged global pivot point
00703 
00704     global_pivot_point = Vector3T(0, 0, 0);
00705 
00706     for (int i = 0; i < n; ++i)
00707     {
00708         global_pivot_point += rotations[i] * local_pivot_point + locations[i];
00709     }
00710 
00711     global_pivot_point /= n;
00712 
00713     // Compute the RMSE
00714 
00715     using std::sqrt;
00716     rmse = T(0);
00717 
00718     for (int i = 0; i < n; ++i)
00719     {
00720         Vector3T noisy_center_point = rotations[i] * local_pivot_point + locations[i];
00721         rmse += (noisy_center_point - global_pivot_point).squaredNorm();
00722     }
00723 
00724     rmse = sqrt(rmse / n);
00725 
00726     return rmse;
00727 }

00911 template <class T>
00912 T PivotCalibrationTwoStep<T>::compute()
00913 {
00914     /*
00915 
00916     Explanation of the algorithm
00917 
00918     As described in the details section, during the calibration procedure,
00919     an object is moved around a pivot point. For example a tool is moved in
00920     such a way that the tool tip remains stationary at a particular position
00921     (which is the pivot point). While moving the object (e.g. the tool) the
00922     object's sensor moves along a circular arc (or in three dimensions on a
00923     calotte of a sphere). Now, in a first step, the center point (= pivot
00924     point) as well as the radius of this sphere are estimated by a simple
00925     least square scheme (that's done in global coordinates).
00926 
00927     Provided the object is still positioned as described above (e.g. the tool
00928     tip still is stationary), using a single measurement (R_i, t_i) the
00929     global pivot point p' can be easily transformed into a local coordinate
00930     p via the relation
00931 
00932         p' = R * p + t    ==>    p = R^(-1) * (p' - t)
00933 
00934     where R is the rotation and t the location of the sensor in the global
00935     coordinate system. However, due to noise, various results for p should
00936     be averaged.
00937 
00938     Implementation details:
00939 
00940     Instead of recording a new set of pairs of (R_i, t_i) after having
00941     determined the virtual sphere, the former measurement is used.
00942 
00943     */
00944 
00945     // Estimate sphere parameters
00946 
00947     FitSphere<T> fitSphere(locations);
00948     fitSphere.compute();
00949     global_pivot_point = fitSphere.getCenterPoint().toArray();
00950 
00951     // Compute the local pivot point
00952 
00953     local_pivot_point = Vector3T(0, 0, 0);
00954 
00955     const int n = rotations.size();
00956 
00957     for (int i = 0; i < n; ++i)
00958     {
00959         Vector3T differenceVectorLocal = rotations[i].inverse() * (global_pivot_point - locations[i]);
00960         local_pivot_point += differenceVectorLocal; // average (part 1)
00961     }
00962 
00963     local_pivot_point = local_pivot_point / n; // average (part 2)
00964 
00965     // Compute the RMSE
00966 
00967     using std::sqrt;
00968     rmse = T(0);
00969 
00970     for (int i = 0; i < n; ++i)
00971     {
00972         Vector3T noisy_center_point = rotations[i] * local_pivot_point + locations[i];
00973         rmse += (noisy_center_point - global_pivot_point).squaredNorm();
00974     }
00975 
00976     rmse = sqrt(rmse / n);
00977 
00978     return rmse;
00979 }
