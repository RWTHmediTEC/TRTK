00287 template <class T>
00288 void PivotCalibration<T>::compute_two_step_procedure()
00289 {
00290     /*
00291 
00292     Explanation of the algorithm
00293 
00294     As described in the details section, during the calibration procedure,
00295     the tool is moved around its tip. In doing so, the tool tip always resides
00296     a the same point, namely the pivot point. At the same time the tool's
00297     sensor moves along the surface of a fictive sphere. Now, in a first step,
00298     the center point (= pivot point) as well as the radius of this sphere are
00299     estimated by a simple least square scheme (in global coordinates).
00300 
00301     Now, the location of the tip in the local sensor coordinate system can
00302     be easily computed from a single measurement (step two). It holds, that a
00303     point p in the local sensor coordinate system corresponds to the point p'
00304     in global coordinates by
00305 
00306         p' = R * p + t
00307 
00308     where R is the rotation and t the location of the sensor in the global
00309     coordinate system. Since the pivot point is known in global coordinates
00310     (let it be p') and the tool tip resides at this location, the location
00311     of the tip in local coordinates is
00312 
00313         p = R^(-1) * (p' - t)
00314     
00315     Using a single measurement (R_i, t_i) and computing the sought translation,
00316     we were actually done. But due to noise, we should take the mean of several
00317     estimated translations.
00318 
00319     */
00320 
00321     // estimate sphere parameters
00322 
00323     FitSphere<T> fitSphere(locations);
00324 
00325     fitSphere.compute();
00326 
00327     Vector3T centerPoint = fitSphere.getCenterPoint().toArray();
00328 
00329     // compute the difference vector, transform it into the sensor system and average it
00330 
00331     translation = Vector3T(0, 0, 0);
00332 
00333     const int n = rotations.size();
00334 
00335     for (int i = 0; i < n; ++i)
00336     {
00337         Vector3T differenceVectorGlobal = centerPoint - locations[i];
00338 
00339         // transform the vector into the local sensor coordinate system
00340 
00341         Vector3T differenceVectorLocal = rotations[i].inverse() * differenceVectorGlobal;
00342 
00343         // average the found translation vector (part 1)
00344 
00345         translation += differenceVectorLocal;
00346     }
00347 
00348     // average the found translation vector (part 2)
00349 
00350     translation = translation / n;
00351 }

00354 template <class T>
00355 void PivotCalibration<T>::compute_combinatorical_approach()
00356 {
00357     /*
00358 
00359     Explanation of the algorithm
00360 
00361     If a tool touches the pivot point with its tip, the pivot point's location M
00362     in global coordinates is
00363 
00364         M = R_i * t + t_i
00365 
00366     where 'R_i' is the rotation and 't_i' the location of the tool sensor in the world or
00367     tracking coordinate system. 't' is the sought location of the tool tip in the sensor's
00368     local coordinate system. Note, that 't' remains the same for all measurements! Now,
00369     using two different measurements, the pivot point M can be eliminated
00370 
00371         R_i * t + t_i = R_j * t + t_j
00372 
00373     This can be rearranged to
00374 
00375         (R_i - R_j) * t = (t_j - t_i)
00376 
00377     Doing this for all combinations of all measurements yields a system of equations
00378     Ax = b (with x = t) which can be solved for the sought translation 't'.
00379 
00380     | R1 - R2   |         | t2   - t1 |
00381     | R1 - R3   |         | t3   - t1 |
00382     | R1 - R4   | * t  =  | t4   - t1 |      <==>    At = b
00383     |   ...     |         |     ...   |
00384     | Rn - Rn-1 |         | tn-1 - tn |
00385     
00386     t = [A^T * A]^(-1) * A^T * B        (Moore–Penrose pseudoinverse)
00387 
00388     Note: Not all combinations are taken into account. For instance, the case i == j
00389           is omitted.
00390 
00391     */
00392 
00393     const int n = rotations.size();
00394     const int max_number_of_combinations = n * (n - 1);
00395     int current_index = 0; // i-th entry (or "row") in A and b
00396 
00397     // Build up A and b.
00398 
00399     MatrixXT A(3 * max_number_of_combinations, 3);
00400     MatrixXT b(3 * max_number_of_combinations, 1);
00401     
00402     for (int i = 0; i < n; ++i)
00403     {
00404         for (int j = 0; j < n; j++)
00405         {
00406             if (i == j || rotations[i].isApprox(rotations[j]))
00407             {
00408                 continue;
00409             }
00410 
00411             A.block(current_index * 3, 0, 3, 3) = rotations[i] - rotations[j];
00412             b.block(current_index * 3, 0, 3, 1) = locations[j] - locations[i];
00413 
00414             ++current_index;
00415         }
00416         
00417     }
00418 
00419     A.resize(current_index * 3, 3);
00420     b.resize(current_index * 3, 1);
00421 
00422     // Solve the system of equations.
00423 
00424     translation = (A.transpose() * A).inverse() * (A.transpose() * b);
00425 }
